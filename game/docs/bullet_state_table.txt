;---
; _BULLET_STATE_NO_BULLET
;
; There is no currently active missile
;                                         
; blank: - 
; draw: -
;
; On fire pressed           -> _BULLET_STATE_NEW
;                               Set missile_x from player x position
;
;                           -> _BULLET_STATE_NO_BULLET
;---

;--
; _BULLET_STATE_NEW
;
; Brand new missile just created
;                                                                 
; blank: - 
; draw: missile
;                    
;                           -> _BULLET_STATE_ACTIVE
;--

;--
; _BULLET_STATE_ACTIVE 
;
; There is an active missile traveling up the screen
;                      
; blank: missile
; draw: missile
;    
; On top of screen          -> _BULLET_STATE_REACHED_TOP_OF_SCREEN
;
; On collision              -> _BULLET_STATE_COLLIDED
;
;                           -> _BULLET_STATE_ACTIVE
;                              Move missile y
;--

;--
; _BULLET_STATE_REACHED_START_OF_SCREEN:
;   
; blank: missile
; draw:  explosion
;
;                           -> _BULLET_STATE_AT_TOP_OF_SCREEN
;                               explosion animation count = 10
;
;--

;--
; _BULLET_STATE_AT_TOP_OF_SCREEN
;
; The missile has reached the top of the screen
;
; blank: missile
; draw:  explosion
;
; Decreased animation count, if 0 -> _BULLET_STATE_DONE_AT_TOP_OF_SCREEN
;                           -> _BULLET_STATE_AT_TOP_OF_SCREEN                 
;
;--

;--
; _BULLET_STATE_DONE_AT_TOP_OF_SCREEN
;
; The explosion of the missile at the top of the screen is complete
; Time to allow a new missile
;
; blank: explosion
; draw: -
;                           -> _BULLET_STATE_NO_BULLET
;--

;--
; _BULLET_STATE_COLLIDED:
;`
; A collision has been detected TODO: Will be revised alien destruction is done
;   blank: missile
;   draw: -
;                           -> _BULLET_STATE_NO_BULLET
;--
